struct LiveRange {
    std::set<int> lines;
    std::set<int> starts;
    std::set<int> ends;
};

struct Web {
    std::string id;
    std::string variableName;
    std::set<int> lines, starts, ends;

    bool overlaps(const LiveRange& other) const;
    void merge(const LiveRange& other);
    std::pair<Web,Web> split() const;
};

struct AllocSettings {
    int         numRegisters = -1;
    std::string  algorithm;    // basic | spilling | splitting | free
    int         algoParam    = -1;
};

T1.2 — Leitura de Dados

Parser

  • Formato: variavel: linha+, linha, linha-
    • + → definição (write)
    • - → último uso (read)
  • Cada linha cria uma LiveRange
  • Fusão de webs por sobreposição:
    • Sem sobreposição → novo web
    • Uma sobreposição → funde nesse web
    • Várias sobreposições → bridging: absorve todos num só
  • Complexidade: O(N × |W| × |L|)
// parser.cpp — lógica de fusão
std::vector<int> overlappingIndices;

for (size_t i = 0; i < allWebs.size(); ++i)
    if (allWebs[i].variableName == currentVariable
        && allWebs[i].overlaps(currentRange))
        overlappingIndices.push_back(i);

if (overlappingIndices.empty()) {
    // Novo web
    Web newWeb;
    newWeb.id = "web" + std::to_string(webCounter++);
    newWeb.variableName = currentVariable;
    newWeb.merge(currentRange);
    allWebs.push_back(newWeb);
} else {
    // Funde na primeira sobreposição
    allWebs[overlappingIndices[0]].merge(currentRange);
    // Absorve restantes (bridging)
    for (int i = overlappingIndices.size()-1; i >= 1; --i)
        allWebs.erase(allWebs.begin() + overlappingIndices[i]);
}
T1.2 — Grafo

Graph<T>

  • Estrutura base do TP reutilizada sem alterações
  • Usada como Graph<string> — cada vértice é o id de um web
  • Grafo não-dirigido: cada interferência adiciona aresta nos dois sentidos
  • As extensões de coloração (color, active) são geridas externamente pelo RegisterAllocator com map<string,int> e map<string,bool>
  • Justificação: sem modificar a estrutura base, os auxiliares permitem simplificação não-destrutiva

Grafo de Interferência

GraphBuilder

  • Dois webs interferem se partilham pelo menos um program point ativo em simultâneo
  • Caso limite — clean handoff: web A termina com - na linha L e web B começa com + na mesma linha L → não interferem
    • Permite reutilizar o mesmo registo em i = i + 1
  • Se todas as linhas comuns são handoffs → sem interferência
  • Complexidade: O(|W|² × |L|)
static bool interferes(const Web& a, const Web& b) {
    std::vector<int> commonLines;
    std::set_intersection(
        a.lines.begin(), a.lines.end(),
        b.lines.begin(), b.lines.end(),
        std::back_inserter(commonLines));

    if (commonLines.empty()) return false;

    for (int line : commonLines) {
        bool a_ends   = (a.ends.count(line) != 0);
        bool b_starts = (b.starts.count(line) != 0);
        bool b_ends   = (b.ends.count(line) != 0);
        bool a_starts = (a.starts.count(line) != 0);

        bool is_clean_handoff =
            (a_ends && b_starts) || (b_ends && a_starts);

        if (!is_clean_handoff) return true;
    }
    return false;
}
T2.1 — Algoritmo Básico

Greedy Graph Colouring

  • Fase 1 — Simplificação: remover repetidamente vértices com grau < K para uma stack
    • Se todos os restantes têm grau ≥ K e não há intervenção → falha total
  • Fase 2 — Coloração: desempilhar e atribuir a menor cor < K não usada pelos vizinhos
  • Se falha → todos os webs vão para memória (registers: 0)
  • Complexidade: O(|V|² + |E|)
// Fase 1 — Simplificação
while (activeCount > 0) {
    bool removedAny = false;
    for (auto v : graph.getVertexSet()) {
        if (active[id] && degrees[id] < settings.numRegisters) {
            active[id] = false;
            stack.push_back(id);
            activeCount--;
            removedAny = true;
            // actualizar graus dos vizinhos
        }
    }
    if (!removedAny) { failed = true; break; }
}

// Fase 2 — Coloração
while (!stack.empty()) {
    std::string id = stack.back(); stack.pop_back();
    std::set<int> usedColours;
    for (auto edge : v->getAdj())
        if (colors[neighbour] != -1)
            usedColours.insert(colors[neighbour]);
    for (int c = 0; c < K; ++c)
        if (!usedColours.count(c)) { colors[id] = c; break; }
}
T2.2 — Web Spilling

Spilling

  • Quando a simplificação fica bloqueada, retira o web de maior grau (atribui à memória) e continua.
  • Permitido até algoParam spills; se exceder → falha.
  • Actualiza os graus dos vizinhos instantaneamente sem reconstruir o grafo.
  • Complexidade: O(|V|² + |E|)
Justificação da Heurística O algoritmo escolhe a variável com o maior grau de interferência (mais vizinhos ativos). A lógica é que o nó com maior grau é o maior bloqueador cromático no grafo. Ao removê-lo para a memória, libertamos a tensão máxima sobre o maior número de vizinhos simultaneamente, maximizando as hipóteses de a simplificação prosseguir. 
else if (settings.algorithm == "spilling"
         && currentSpills < settings.algoParam) {

    // Selecção heurística — maior grau
    for (auto v : graph.getVertexSet()) {
        if (active[id] && degrees[id] > maxMetric) {
            maxMetric = degrees[id];
            problematicNode = id;
        }
    }

    // Desactivar — atribuído à memória (color = -1)
    active[problematicNode] = false;
    activeCount--;
    currentSpills++;

    // Actualizar vizinhos

        if (active[neighbour])
            degrees[neighbour]--;
}
T2.3 — Web Splitting

Splitting

  • Em vez de descartar, divide o web problemático em dois sub-webs pela metade cronológica.
  • Devolve sinal "__RESTART__" → pipeline reconstrói o grafo de interferência.
  • Permitido até algoParam divisões.
  • Complexidade: O(K × (|W|²·|L| + |V|² + |E|))
Justificação da Heurística O alvo do split é novamente o web com o maior grau. Ao cortar cronologicamente o live range mais intercetado, criamos duas variáveis distintas. Matematicamente, os vizinhos da primeira metade deixam de interferir com os vizinhos da segunda metade, diminuindo drasticamente o grau local daquela região e viabilizando a coloração sem perda de registos. 
if (settings.algorithm == "splitting"
    && currentSplits < settings.algoParam) {

    std::vector<Web> newWebs;
    for (const Web& w : webs) {
        if (w.id == problematicNode) {
            auto [w1, w2] = w.split();
            newWebs.push_back(w1);
            newWebs.push_back(w2);
        } else {
            newWebs.push_back(w);
        }
    }
    webs = newWebs;
    currentSplits++;
    return {{"__RESTART__", 1}}; // sinaliza rebuild
}
T2.4 — Algoritmo Livre

Free Mode

  • Modo de alocação agressiva: spilling sem limite de variáveis.
  • Garante sempre uma alocação válida (no limite, tudo na memória).
  • Complexidade: O(|V|² + |E|)
Justificação da Heurística de Spill Em vez do grau, usamos uma heurística temporal: selecionamos o web com o maior footprint (maior número de linhas ativas). Num modo sem limites restritos, varrer para a memória a variável que está viva durante mais tempo é altamente eficiente, pois essa variável cruza o tempo de vida de dezenas de outras variáveis locais menores, libertando instantaneamente múltiplas pressões cruzadas. 
if (settings.algorithm == "free") {
    // Heurística: maior número de linhas ativas
    for (auto v : graph.getVertexSet()) {
        if (!active[id]) continue;
        int lineCount = 0;
        for (const Web& w : webs)
            if (w.id == id)
                lineCount = w.lines.size();
        if (lineCount > maxMetric) {
            maxMetric = lineCount;
            problematicNode = id;
        }
    }
}
// Spill sem verificar limite
active[problematicNode] = false;
activeCount--; currentSpills++;

Reflexão

Dificuldades

A deteção correta de interferência nos limites dos webs foi o aspeto mais subtil: dois webs que partilham uma linha apenas por clean handoff (um termina com - e outro começa com + na mesma linha) não interferem — implementar esta regra exigiu cuidado especial em GraphBuilder::interferes().

A fusão transitiva de webs no parser (bridging): uma nova live range pode ligar dois webs previamente separados da mesma variável, obrigando a absorver múltiplos webs de uma vez e a iterar os índices de trás para a frente para não corromper o vector.

A integração do mecanismo de restart para splitting ("__RESTART__") entre RegisterAllocator e main.cpp foi o aspeto mais desafiante da arquitectura — o alocador modifica o vector de webs por referência e sinaliza ao pipeline para reconstruir o grafo.
